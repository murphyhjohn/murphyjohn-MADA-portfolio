# summary stats by hand
format_ci <- function(df, d = 2) {
    fmt_str <- paste0("%.", d, "f")
    out <- paste0(
        sprintf(fmt_str, df$y), " (",
        sprintf(fmt_str, df$ymin), ", ",
        sprintf(fmt_str, df$ymax), ")"
    )
    
    return(out)
}
